Project 3

Personal finances are an important part of life. Sometimes you need some extra money and decide to take a loan, or you want to buy a house using a mortgage. To make an informed decision, you need to be able to calculate different financial parameters. I have made a program that can help you make these informed decisions!

Code

class Category:

def __init__(self, name):
self.name = name
self.ledger = []
self.balance = 0

def receipt(self):
# maybe i can use len() - 30 somehow?
#header = (13 * "*" + self.name + 13 * "*")
header = self.name.center(30, "*")
print(header)
for item in self.ledger:
desc = item["description"]
amount = "%.2f" % item["amount"]
# I neet to left align desc and right align amount automatically
txt = "{:<23}{:<10}\n".format(desc, amount)
print(txt)
# should show the total amount
print("Total: " + '%.2f' % self.balance)

def deposit(self, amount, description=""):
self.ledger.append({"amount": amount, "description": description})
self.balance += amount

def withdraw(self, amount, description=""):
if self.check_funds(amount):
self.ledger.append({"amount": -amount, "description": description})
self.balance -= amount
return True
else:
return False

def get_balance(self):
return self.balance

def transfer(self, amount, instance):
isworking = self.withdraw(
amount, description="Tansfer to " + instance.name)
if not isworking:
return False
instance.deposit(amount, description="Tansfer from " + self.name)

def check_funds(self, amount):
if amount <= self.balance:
return True
else:
return False


food = Category("Food")
clothing = Category("Clothing")
entertainment = Category("Entertainment")
