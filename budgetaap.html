Budget App

This is a budget application backend created in Python. It shows information such as deposits, withdraws, current balance, and expenses as well as the total balance in each category created.

A user can create as many categories as he/she wants. The program will print a receipt for each category created along with the transaction details. Also a spenditure chart is also created showing a percentage spent for each category created.

Code


class Category:

def __init__(self, name):
self.name = name
self.ledger = list()

def __str__(self):
category = f"{self.name:*^30}\n"
items = ""
total = 0
for item in self.ledger:
items += f"{item['description'][0:23]:23}" + \
f"{item['amount']:>7.2f}" + '\n'
total += item['amount']
output = category + items + "Total: " + str(round(total, 2))
return output

def deposit(self, amount, description=""):
self.deposit = {'amount': amount, "description": description}
self.ledger.append(self.deposit)

def withdraw(self, amount, description=""):

if self.check_funds(amount):
self.ledger.append({'amount': -amount, 'description': description})
return True
return False

def get_balance(self):

balance = 0
for entry in self.ledger:
balance += entry['amount']
return balance

def transfer(source, amount, destination):

if source.check_funds(amount):
source.withdraw(amount, "Transfer to " + destination.name)
destination.deposit(amount, "Transfer from " + source.name)
return True
return False

def check_funds(category, amount):

if (category.get_balance() >= amount):
return True
return False

def get_withdrawals(self):

total = 0
for item in self.ledger:
if item['amount'] < 0:
total += item['amount']
return round(total, 2)


def truncate(n):
multiplier = 10
return int(n * multiplier)/multiplier


def getTotals(categories):
total = 0
breakdown = []
for category in categories:
total += category.get_withdrawals()
breakdown.append(category.get_withdrawals())
rounded = list(map(lambda x: truncate(x/total), breakdown))
return rounded


def create_spend_chart(categories):

res = "Percentage spent by category\n"
i = 100
totals = getTotals(categories)

while i >= 0:
cat_spaces = " "
for total in totals:
if total * 100 >= i:
cat_spaces += "o "
else:
cat_spaces += " "
res += str(i).rjust(3) + "|" + cat_spaces + ("\n")
i -= 10

dashes = "-" + "---" * len(categories)
names = []
x_axis = ""
for category in categories:
names.append(category.name)

maxi = max(names, key=len)

for x in range(len(maxi)):
nameStr = ' '

for name in names:
if x >= len(name):
nameStr += " "
else:
nameStr += name[x] + " "

if (x != len(maxi) - 1):
nameStr += '\n'

x_axis += nameStr

res += dashes.rjust(len(dashes)+4) + "\n" + x_axis
return res


food = Category("Food")
food.deposit(1000, "initial deposit")
food.withdraw(10.15, "groceries")
food.withdraw(15.89, "restaurant and more food for dessert")
print(food.get_balance())
clothing = Category("Clothing")
food.transfer(50, clothing)
clothing.deposit(300, "paycheck")
clothing.withdraw(25.55)
clothing.withdraw(100)
auto = Category("Auto")
auto.deposit(1000, "initial deposit")
auto.withdraw(15)

print(food)
print(clothing)

print(create_spend_chart([food, clothing, auto]))
